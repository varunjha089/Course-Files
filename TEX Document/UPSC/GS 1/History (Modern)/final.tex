% UPSC-style answer sheet frame
\newcommand{\drawupscframe}[1]{
\begin{tikzpicture}[remember picture, overlay]
    % Header line
    \draw[line width=1pt]
        ([yshift=-3cm]current page.north west) -- 
        ([yshift=-3cm]current page.north east);

    % Left vertical margin line
    \draw[line width=1pt]
        ([xshift=1in]current page.south west) -- 
        ([xshift=1in]current page.north west);

    % Right vertical margin line
    \draw[line width=1pt]
        ([xshift=-1in]current page.south east) -- 
        ([xshift=-1in]current page.north east);

    % Header Text
    \node[anchor=north, font=\bfseries\Large, align=center] 
        at ([yshift=-1.2cm]current page.north) 
        {From Plassy to Partition\\Ch - 0#1};
\end{tikzpicture}
}

% Command to add 44 answer pages for a chapter
\newcommand{\addanswerpages}[1]{%
    \foreach \i in {1,...,44} {
        \drawupscframe{#1}
        \vspace{3cm}
        \vfill
        \newpage
    }
}
